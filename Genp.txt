## Step 2: Add Password Entry Class

Create `src/main/java/io/erikka/vesta/model/PasswordEntry.java`:

```java
package io.erikka.vesta.model;

import javafx.beans.property.SimpleStringProperty;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;

public class PasswordEntry {
    private final SimpleStringProperty time;
    private final SimpleStringProperty label;
    private final SimpleStringProperty type;
    private final SimpleStringProperty password;
    private final LocalDateTime timestamp;

    public PasswordEntry(String label, String type, String password) {
        this.timestamp = LocalDateTime.now();
        DateTimeFormatter formatter = DateTimeFormatter.ofPattern("HH:mm:ss");
        this.time = new SimpleStringProperty(this.timestamp.format(formatter));
        this.label = new SimpleStringProperty(label);
        this.type = new SimpleStringProperty(type);
        this.password = new SimpleStringProperty(password);
    }

    public String getTime() { return time.get(); }
    public String getLabel() { return label.get(); }
    public String getType() { return type.get(); }
    public String getPassword() { return password.get(); }
    public LocalDateTime getTimestamp() { return timestamp; }

    public SimpleStringProperty timeProperty() { return time; }
    public SimpleStringProperty labelProperty() { return label; }
    public SimpleStringProperty typeProperty() { return type; }
    public SimpleStringProperty passwordProperty() { return password; }
}
```

@FXML
private void handleGeneratePassword() {
    passwordProgressIndicator.setVisible(true);
    generatedPasswordArea.clear();
    passwordStrengthLabel.setText("");

    new Thread(() -> {
        try {
            String password = generatePasswordByType();

            Platform.runLater(() -> {
                passwordProgressIndicator.setVisible(false);
                generatedPasswordArea.setText(password);
                updatePasswordStrength(password);
            });

        } catch (Exception e) {
            Platform.runLater(() -> {
                passwordProgressIndicator.setVisible(false);
                showError("Password generation failed: " + e.getMessage());
            });
        }
    }).start();
}

private String generatePasswordByType() {
    if (customPasswordRadio.isSelected()) {
        return generateCustomPassword();
    } else if (base64PasswordRadio.isSelected()) {
        return generateBase64Password();
    } else if (base32PasswordRadio.isSelected()) {
        return generateBase32Password();
    } else if (hexPasswordRadio.isSelected()) {
        return generateHexPassword();
    } else if (uuidPasswordRadio.isSelected()) {
        return java.util.UUID.randomUUID().toString();
    }
    throw new IllegalStateException("No password type selected");
}

private String generateCustomPassword() {
    int length = passwordLengthSpinner.getValue();
    StringBuilder charset = new StringBuilder();

    if (includeUppercaseCheck.isSelected()) {
        charset.append("ABCDEFGHIJKLMNOPQRSTUVWXYZ");
    }
    if (includeLowercaseCheck.isSelected()) {
        charset.append("abcdefghijklmnopqrstuvwxyz");
    }
    if (includeNumbersCheck.isSelected()) {
        charset.append("0123456789");
    }
    if (includeSymbolsCheck.isSelected()) {
        String symbols = useCustomSymbolsCheck.isSelected() && !customSymbolsField.getText().isBlank()
            ? customSymbolsField.getText()
            : DEFAULT_SYMBOLS;
        charset.append(symbols);
    }

    if (charset.length() == 0) {
        throw new IllegalArgumentException("At least one character type must be selected");
    }

    java.security.SecureRandom random = new java.security.SecureRandom();
    StringBuilder password = new StringBuilder(length);

    for (int i = 0; i < length; i++) {
        int index = random.nextInt(charset.length());
        password.append(charset.charAt(index));
    }

    return password.toString();
}

private String generateBase64Password() {
    int length = passwordLengthSpinner.getValue();
    java.security.SecureRandom random = new java.security.SecureRandom();
    byte[] bytes = new byte[length * 3 / 4]; // Adjust for Base64 expansion
    random.nextBytes(bytes);
    String base64 = java.util.Base64.getUrlEncoder().withoutPadding().encodeToString(bytes);
    return base64.substring(0, Math.min(length, base64.length()));
}

private String generateBase32Password() {
    int length = passwordLengthSpinner.getValue();
    String base32Chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ234567";
    java.security.SecureRandom random = new java.security.SecureRandom();
    StringBuilder password = new StringBuilder(length);

    for (int i = 0; i < length; i++) {
        password.append(base32Chars.charAt(random.nextInt(base32Chars.length())));
    }

    return password.toString();
}

private String generateHexPassword() {
    int length = passwordLengthSpinner.getValue();
    java.security.SecureRandom random = new java.security.SecureRandom();
    byte[] bytes = new byte[length / 2 + 1];
    random.nextBytes(bytes);
    StringBuilder hex = new StringBuilder();
    for (byte b : bytes) {
        hex.append(String.format("%02x", b));
    }
    return hex.substring(0, length);
}

private void updatePasswordStrength(String password) {
    int score = calculatePasswordStrength(password);
    String strength;
    String color;

    if (score < 3) {
        strength = "Weak";
        color = "#e53e3e";
    } else if (score < 4) {
        strength = "Medium";
        color = "#dd6b20";
    } else if (score < 5) {
        strength = "Strong";
        color = "#38a169";
    } else {
        strength = "Very Strong";
        color = "#2f855a";
    }

    passwordStrengthLabel.setText("Strength: " + strength);
    passwordStrengthLabel.setStyle("-fx-text-fill: " + color + "; -fx-font-weight: bold;");
}

private int calculatePasswordStrength(String password) {
    int score = 0;

    if (password.length() >= 8) score++;
    if (password.length() >= 12) score++;
    if (password.length() >= 16) score++;
    if (password.matches(".*[A-Z].*")) score++;
    if (password.matches(".*[a-z].*")) score++;
    if (password.matches(".*\\d.*")) score++;
    if (password.matches(".*[!@#$%^&*()_+\\-=\\[\\]{}|;:,.<>?].*")) score++;

    return Math.min(score, 7);
}

@FXML
private void handleSaveToCredFile() {
    String password = generatedPasswordArea.getText();
    if (password == null || password.isBlank()) {
        showError("No password generated to save");
        return;
    }

    String label = passwordLabelField.getText();
    if (label == null || label.isBlank()) {
        showError("Please enter a password label");
        return;
    }

    String credFilePath = credFilePathField.getText();
    if (credFilePath == null || credFilePath.isBlank()) {
        showError("Please select a credential file location");
        return;
    }

    try {
        saveToCredentialFile(credFilePath, label, password);
        addToRecentPasswords(label, password);
        showInfo("Password saved to credential file!");
        passwordLabelField.clear();
    } catch (IOException e) {
        showError("Failed to save to file: " + e.getMessage());
    }
}

@FXML
private void handleGenerateAndSave() {
    handleGeneratePassword();
    // Wait a bit for generation to complete, then save
    new Thread(() -> {
        try {
            Thread.sleep(500); // Wait for generation
            Platform.runLater(this::handleSaveToCredFile);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
    }).start();
}

private void saveToCredentialFile(String filePath, String label, String password) throws IOException {
    File file = new File(filePath);

    // Create file if it doesn't exist
    if (!file.exists()) {
        file.getParentFile().mkdirs();
        file.createNewFile();
    }

    // Format timestamp
    LocalDateTime now = LocalDateTime.now();
    DateTimeFormatter formatter = DateTimeFormatter.ofPattern("dd MMMM, yyyy - h:mma", Locale.ENGLISH);
    String timestamp = now.format(formatter);

    // Prepare entry
    String entry = String.format("%n# %s%n%s=%s%n", timestamp, label, password);

    // Append to file
    Files.write(
        file.toPath(),
        entry.getBytes(StandardCharsets.UTF_8),
        StandardOpenOption.CREATE,
        StandardOpenOption.APPEND
    );
}

private void addToRecentPasswords(String label, String password) {
    String type = getSelectedPasswordType();
    PasswordEntry entry = new PasswordEntry(label, type, password);
    recentPasswords.add(0, entry); // Add to beginning

    // Limit to 50 entries
    if (recentPasswords.size() > 50) {
        recentPasswords.remove(50, recentPasswords.size());
    }
}

private String getSelectedPasswordType() {
    if (customPasswordRadio.isSelected()) return "Custom";
    if (base64PasswordRadio.isSelected()) return "Base64";
    if (base32PasswordRadio.isSelected()) return "Base32";
    if (hexPasswordRadio.isSelected()) return "Hex";
    if (uuidPasswordRadio.isSelected()) return "UUID";
    return "Unknown";
}

        saveToCredentialFile(credFilePath, selected.getLabel(), selected.getPassword());
        showInfo("Password saved to credential file!");
    } catch (IOException e) {
        showError("Failed to save to file: " + e.getMessage());
    }
}

@FXML
private void handleClearPasswordHistory() {
    recentPasswords.clear();
    showInfo("Password history cleared!");
}
